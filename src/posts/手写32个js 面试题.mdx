3. 手写es6 filter方法

   ```javascript
   Array.prototype.filter = function (callback, thisArg) {
      const result = []
      const list = Object(this)
      const len = this.length >>> 0 // 必须为正整数
      for (let i = 0; i < len; i++) {
         if (i in list) {
            if (callback.call(thisArg, list[i], i, list)) {
               result.push(list[i])
            }
         }
      }
   }
   ```

4. 手写es6 map方法

   ```javascript
   Array.prototype.map = function (callback, thisArg) {
      const result = [];
      const list = Object(this);
      const len = this.length >>> 0; //  同上
      for (let i = 0; i < len; i++) {

         // 调用回调， 将新的值传回去
         if (i in list) {
            result[i] = callback.call(thisArg, list[i], i, list)
         }
      }
   }
   ```
